int a;
int b;
int c;

read(a);
read(b);
read(c);

if (a + b > c && a + c > b && b + c > a) then
    if (a == b && b == c) then
        write("Triangulo equilatero");
    else
        if (a == b || a == c || b == c) then
            write("Triangulo isosceles");
        else
            write("Triangulo escaleno");
        endif;
    endif;
else
    write("Invalid side lengths");
endif;
